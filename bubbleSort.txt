        public static int[] BubbleSort(int[] numeros)
        {
            int n = numeros.Length;            

            for (var i = 0; i < n; i++)
            {
                for (var j = 0; j < (n - 1 - i); j++)
                {
                    var sort = numeros[j] < numeros[j + 1];
                    if (sort)
                    {
                        int temp = numeros[j];
                        numeros[j] = numeros[j + 1];
                        numeros[j + 1] = temp;
                    }
                }
            }
            return numeros;
        }
